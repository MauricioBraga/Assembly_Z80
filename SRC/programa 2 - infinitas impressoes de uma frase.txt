CHPUT:		equ 0x00a2   ; imprime 1 caractere na tela
CLS:		equ 0x00c3   ; limpa a tela

org 0xD000   ; Endere√ßo de inicio do programa

start:
		xor a
		call CLS	; limpa a tela

inicio:		ld hl, message
mainLoop:
		ld a, (hl)
        	cp 0
            	jr z, inicio
            	call CHPUT
            	inc hl
            	jr mainLoop

message:    	db "Ola Mundo!",13,10,0

end start
